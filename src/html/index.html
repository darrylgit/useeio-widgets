Examples

Impact heatmap 
                
                    var heatmap = useeio.ImpactHeatmap.on({
                        webapi: {
                            endpoint: './api',
                            model: 'USEEIOv1.2',
                            asJsonFiles: true,
                        },
                        selector: '.impact-heatmap',
                    });
Impact chart 
                
                    var impactChart = useeio.ImpactChart.on({
                        webapi: {
                            endpoint: './api',
                            model: 'USEEIOv1.2',
                            asJsonFiles: true,
                        },
                        selector: '.impact-chart',
                        columns: 3,
                        width: 800,
                        height: 300,
                    });
                    impactChart.update({
                        sectors: ["312120", "31151A", "312110", "312130"],
                    });
Sector list with hash config 
                
                    var config = useeio.HashConfigTransmitter.create();
                    var sectorList = useeio.SectorList.on({
                        webapi: {
                            endpoint: './api',
                            model: 'USEEIOv1.2',
                            asJsonFiles: true,
                        },
                        selector: '.sector-list',
                    });
                    config.join(sectorList);
Sector list with linked impact chart 
                
                    var webapi = {
                        endpoint: './api',
                        model: 'USEEIOv1.2',
                        asJsonFiles: true,
                    };
                    var sectorList = useeio.SectorList.on({
                        webapi: webapi,
                        selector: '.sector-list',
                    });
                    var impactChart = useeio.ImpactChart.on({
                        webapi: webapi,
                        selector: '.impact-chart',
                    });
                    var config = useeio.HashConfigTransmitter.create();
                    config.join(sectorList);
                    config.join(impactChart);
Filter widget 
                
                    var filter = useeio.FilterWidget.on({
                        selector: '#filter-widget',
                    });
                    var config = useeio.HashConfigTransmitter.create();
                    config.join(filter);
