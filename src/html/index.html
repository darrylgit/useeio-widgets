Examples

Impact heatmap 
                
                    var heatmap = useeio.ImpactHeatmap.on({
                        endpoint: '/api',
                        model: 'USEEIOv1.2',
                        selector: '.impact-heatmap',
                    });
Impact chart 
                
                    var impactChart = useeio.ImpactChart.on({
                        endpoint: '/api',
                        model: 'USEEIOv1.2',
                        selector: '.impact-chart',
                        columns: 3,
                        width: 800,
                        height: 300,
                    });
                    impactChart.update({
                        sectors: ["312120", "31151A", "312110", "312130"],
                    });
Sector list with hash config 
                
                    var config = useeio.HashConfigTransmitter.create();
                    var sectorList = useeio.SectorList.on({
                        endpoint: '/api',
                        model: 'USEEIOv1.2',
                        selector: '.sector-list',
                    });
                    config.join(sectorList);
Sector list with linked impact chart 
                
                    var sectorList = useeio.SectorList.on({
                        endpoint: '/api',
                        model: 'USEEIOv1.2',
                        selector: '.sector-list',
                    });
                    var impactChart = useeio.ImpactChart.on({
                        endpoint: '/api',
                        model: 'USEEIOv1.2',
                        selector: '.impact-chart',
                    });
                    var config = useeio.HashConfigTransmitter.create();
                    config.join(sectorList);
                    config.join(impactChart);
